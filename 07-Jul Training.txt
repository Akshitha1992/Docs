pipeline {
    agent any
    stages{
        stage('Build') {
           steps {
                echo 'Building'
                sh '''
                echo "by the way i can do multiple works"
                ls -lah
                '''
                timeout(time:3, unit:'SECONDS'){
                    sh 'sleep 5'
                }
            }
        }
    
        stage('test'){
            steps {
                echo 'testing...'
            }
        }
        stage('Deploy') {
            steps {
                echo 'deploying'
            }
        }
}
}
